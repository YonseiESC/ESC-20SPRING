# scipy stats package does not allow for plotting pdf-theta so we need to define formula directly

def likelihood(D, p):
    N = D.size
    suc = D.sum()
    theta = np.linspace(0,1,100)
    return p**suc * (1-p)**(N-suc)
# note that we neglected the constant as it will be cancelled out in applying Bayes Rule

def likelihood(D, x) :
    mu = np.mean(D)
    sigmasq= np.var(D)
    return np.exp(((x-mu)**2)/(-2*sigmasq))
