#Prior
mu_0,lmda_mu_0=15,0.5

prior=norm(15,1/lmda_mu_0)
theta=np.linspace(0,30,100)
plt.plot(theta,prior.pdf(theta),color='r')
plt.title('Prior on mu (mu=15, lmda_mu=0.5)')
plt.xlabel('mu')
plt.axvline(15,color='b',linestyle='--')
plt.show()

#Likelihood
def loglikelihood(D,mu):
    l=0
    for x in D:
        l+=np.log(norm.pdf(x,mu,1))
    return l

plt.plot(theta,loglikelihood(data,theta),color='g')
plt.title('log likelihood')
plt.xlabel('mu')
plt.ylabel('log_p(data|mu)')
plt.axvline(xbar,color='b',linestyle=':')
plt.show()

#Posterior
lmda_mu_pos = lmda_mu_0 + lmda
mu_pos = (mu_0 * lmda_mu_0 + xbar * lmda)/lmda_mu_pos

posterior=norm(mu_pos,1/lmda_mu_pos)
theta=np.linspace(0,30,100)

plt.plot(theta,posterior.pdf(theta),color='r')
plt.title('Posterior on mu (mu=15,xbar=20.2)')
plt.xlabel('mu')
plt.axvline(15,color='darkblue',linestyle='--')
plt.axvline(mu_pos,color='b',linestyle='-')
plt.axvline(xbar,color='skyblue',linestyle=':')
plt.show()
